\begin{frame}\frametitle{Профиль показателя преломления}
  Для расчета профиля показателя в физическом пространстве воспользуемся формулой:
  \begin{equation}
    n_z =
    \left|\frac{dw}{dz}\right| n_w =
    \frac{2}{(x+1)^2+y^2}
    \label{eq:pok_prelom}
  \end{equation}

\end{frame}
